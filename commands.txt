python -m django startproject <имя> - создает базовую конфигаруцию проекта
python manage.py runserver - Запускает django-сервер
python manage.py startapp <имя приложения> - создает новое приложение
python manage.py check - проверяет наш проект на предмет наличия ошибок
python manage.py makemigration - команда не создает новые таблицы в БД, она создает инструкции для БД, чтобы создать новые таблицы
python manage.py migrate - применяет последнюю миграцию к БД
winget install --id Git.Git -e --source winget - устанока Git
python -m ensurepip - установка pip
pip install django - установка django
python manage.py shell - вызов консоли django
